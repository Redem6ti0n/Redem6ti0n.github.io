from PIL import Image

filename = r'2.png'
img = Image.open(filename)
size = img.size
print(size)

# 准备将图片切割成9张小图片
weight = int(size[0] // 29)
height = int(size[1] // 29)
# 切割后的小图的宽度和高度
print(weight, height)

for j in range(29):
    for i in range(29):
        box = (weight * i, height * j, weight * (i + 1), height * (j + 1))
        region = img.crop(box)
        region.save('{}{}.png'.format(j, i))
