function strToHex($str){
    $hex="";
    for($i=0;$i<strlen($str)-1;$i++) {
        $len = strlen(dechex(ord($str[$i])));
        if ($len === 1) {
            $hex = $hex . "%0" . dechex(ord($str[$i]));
        } else {
            $hex = $hex . "%" . dechex(ord($str[$i]));
        }
    }
    return $hex;
}
